import tkinter as tk from tkinter import messagebox import math try: import matplotlib.pyplot as plt except ImportError: messagebox.showerror( "Error", "Matplotlib no está instalado.\n\nEjecute:\npy -m pip install matplotlib" ) raise def calcular_y_graficar(): try: D = float(txt_diametro.get()) h = float(txt_altura.get()) x0 = float(txt_x0.get()) xcrit = float(txt_xcrit.get()) icorr_pas = float(txt_pasivado.get()) icorr_mod = float(txt_moderado.get()) icorr_sev = float(txt_severo.get()) if x0 <= xcrit: raise ValueError( "El espesor inicial debe ser mayor que el crítico." ) area = math.pi * D * h k = 0.0116 vp = icorr_pas * k vm = icorr_mod * k vs = icorr_sev * k vida_pas = (x0 - xcrit) / vp vida_mod = (x0 - xcrit) / vm vida_sev = (x0 - xcrit) / vs resultado.set( f"Área húmeda = {area:.2f} m²\n\n" f"Vida útil Pasivado = {vida_pas:.2f} años\n" f"Vida útil Moderado = {vida_mod:.2f} años\n" f"Vida útil Severo = {vida_sev:.2f} años" ) tiempo = list(range(101)) esp_pas = [x0 - vp * t for t in tiempo] esp_mod = [x0 - vm * t for t in tiempo] esp_sev = [x0 - vs * t for t in tiempo] plt.figure(figsize=(10, 6)) plt.plot( tiempo, esp_pas, linewidth=3, label="Pasivado" ) plt.plot( tiempo, esp_mod, linewidth=3, label="Moderado" ) plt.plot( tiempo, esp_sev, linewidth=3, label="Severo" ) plt.axhline( y=xcrit, linestyle="--", linewidth=2, label="Espesor crítico" ) plt.title("Predicción de Vida Útil del Tanque") plt.xlabel("Tiempo (años)") plt.ylabel("Espesor (mm)") plt.grid(True) plt.legend() plt.show() except ValueError as e: messagebox.showerror( "Dato inválido", str(e) ) except Exception as e: messagebox.showerror( "Error", f"Ocurrió un error:\n{e}" ) ventana = tk.Tk() ventana.title("Predictor de Vida Útil de Tanques") ventana.geometry("650x700") ventana.configure(bg="#D6EAF8") titulo = tk.Label( ventana, text="PREDICTOR DE VIDA ÚTIL DE TANQUES", bg="#1F618D", fg="white", font=("Arial", 14, "bold"), pady=10 ) titulo.pack(fill="x") def crear_campo(texto): tk.Label( ventana, text=texto, bg="#D6EAF8", font=("Arial", 10, "bold") ).pack() entrada = tk.Entry( ventana, width=25 ) entrada.pack(pady=3) return entrada txt_diametro = crear_campo("Diámetro del tanque (m)") txt_altura = crear_campo("Altura del líquido (m)") txt_x0 = crear_campo("Espesor inicial x0 (mm)") txt_xcrit = crear_campo("Espesor crítico xcrit (mm)") txt_pasivado = crear_campo("icorr Pasivado (µA/cm²)") txt_moderado = crear_campo("icorr Moderado (µA/cm²)") txt_severo = crear_campo("icorr Severo (µA/cm²)") tk.Button( ventana, text="CALCULAR Y GRAFICAR", command=calcular_y_graficar, bg="#27AE60", fg="white", font=("Arial", 11, "bold") ).pack(pady=15) resultado = tk.StringVar() tk.Label( ventana, textvariable=resultado, bg="#FCF3CF", justify="left", font=("Arial", 11), padx=15, pady=15 ).pack(pady=10) ventana.mainloop()
